56template <typename T = double>
57std::vector<T> linspace(T a, T b, size_t N)
58{
59 T h = (b - a) / static_cast<T>(N - 1);
60 std::vector<T> xs(N);
61 typename std::vector<T>::iterator x;
62 T val;
63 for (x = xs.begin(), val = a; x != xs.end(); ++x, val += h)
64 *x = val;
65 return xs;
66};
